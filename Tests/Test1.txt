
/*Выберите из таблицы Водители записи, где стаж водителя не равен 3, 5, 6, 10 и  место прописки 'eee'.*/

SELECT *
FROM [Водители]
WHERE [Стаж водителя] NOT IN (3,5,6,10)
AND [Место прописки] = 'eee'

/*Найдите № паспортов водителей, которые ни разу не совершали поездки (решите задачу через left join).*/

SELECT В.[№ паспортов]
FROM [Водители] В
LEFT JOIN [Диспетчерская] Д
ON В.[Код водителя] = Д. [Код водителя]
WHERE Д. [Код водителя] IS NULL


/*Написать запрос на выборку авто с годом выпуска = 2015 и его маршруты (таблицы Aвто, Диспеческая, Маршрут). 
Вывести рег номер авто, время прибытия и отбытия, пункт назначения.*/

SELECT А.[Регистрационной № авто], Д.[Время прибытия], Д.[Время отбытия], М.[Пункт назначения]
FROM [Автомобили] А
JOIN [Диспетчерская] Д
ON А.[Код авто] = Д. [Код авто]
JOIN [Маршрут] М
ON Д.[Путевка] = М.[Код маршрута]
WHERE [Год выпуска авто] = 2015
